\section{Qu'est-ce que PSTricks}
\frame{\frametitle{\textbf{\huge{Qu'est-ce que {\color{red}PSTricks}}?}}
\begin{itemize}
\item {\color{red}PSTricks} est une extension de LaTeX qui permet d'utiliser la majeure partie des possibilités de {\color{red}PostScript}.\pause
\item {\color{red}PSTricks} est un outil de préparation de dessins.\pause
\item {\color{red}PSTricks} est ensemble de commandes directement utilisables depuis {\color{red}LATEX}.\pause
\item {\color{red}PSTricks} peut être chargé via 
\end{itemize}
 \begin{block}{\textbf{Code source :}}
 $\backslash usepackage \{pst-all \}$
 \end{block}
}

\section{Origine de syst\`eme de coordonnées}
\frame{\frametitle{\textbf{\huge{Origine de syst\`eme de coordonnées}}}
L'origine du système de coordonnées utilisés par {\color{red}PSTricks} est le point courant de {\color{red} Latex}

\begin{block}{\textbf{Code source :}}
Bonjour ${\color{Green}\backslash psline(2,1)}$ 
\end{block}
\begin{block}{\textbf{Résultat :}}
Bonjour \psline(2,1) 
\end{block}
\pause

\begin{block}{\textbf{Code source :}}
Bonjour ${\color{Green}\backslash psline(2,1)}$ ${\color{Green}\backslash psline(1,1)}$
\end{block}
\begin{block}{\textbf{Résultat :}}
Bonjour \psline(2,1) \psline(1,1) 
\end{block}
}

\section{Réserver de l'espace à un dessin}
\frame{\frametitle{\textbf{\huge{Réserver de l'espace à un dessin}}}

Pour réserver de l'espace à un dessin, on utilisera
l'environnement {\color{red}\textbf{pspicture}} :
\begin{block}{\textbf{Syntaxe}}
$\backslash begin{\color{red}\{pspicture\}} {\color{orange}(x_0 , y_0 )(x_1 , y_1 )}\\
\backslash end {\color{red}\{pspicture\}}$
\end{block}
\pause
où 
\begin{itemize}
\item ${\color{orange}(x_0, y_0)}$ est le point bas gauche.
\item ${\color{orange}(x_1 , y_1)}$ est le point haut droit
du rectangle délimitant la figure.
\end{itemize}
\pause
\begin{block}{\textbf{Remarque :}}
Si $(x_0 , y_0 )$ est omis, il est remplacé par $(0, 0)$.
\end{block} 
}
% % % % % % % % % % % % % % % % % % % % % % % % % % % % % % 
\subsection{Exemple}
\frame{\frametitle{}
\begin{block}{\textbf{Code source :}}
Bonjour\\
$\backslash begin{\color{red}\{pspicture\}} (0 , 0 )(2.1 , 1.1 )\\
{\color{Green}\backslash psline}{\color{black}(2,1)}\\
{\color{black}\backslash end} {\color{red}\{pspicture\}}$\\
{\color{black}les amis!}
\end{block}
\begin{block}{\textbf{Résultat :}}
Bonjour
\begin{pspicture}(0,0)(2.1,1.1)
\psline(2,1)
\end{pspicture}
les amis!
\end{block}
}

\section{Unité}
\frame{\frametitle{\textbf{\huge{Unité :}}}
L'unité graphique de {\color{red}Pstricks} est {\color{red}$1$ cm} par défaut. 
\begin{itemize}
\item Si on veut la changer, on écrit (par exemple) avant le $\backslash begin{\color{red}\{pspicture \}}$ :\\
\begin{block}{\textbf{Syntaxe}}
${\color{Green}\backslash psset}\{{\color{Green}xunit}=2cm,{\color{Green}yunit}=0.5cm\}$
\end{block}
\pause
\item Si les {\color{red}{deux unités graphiques sont les mêmes}}, on peut
écrire :
\begin{block}{\textbf{Syntaxe}}
${\color{Green}\backslash psset}\{ {\color{Green}unit}=2cm\}$
\end{block}
\pause
\item Si {\color{red}{toutes les lignes ont pour épaisseur $2.5$ pt}}, on peut écrire :
\begin{block}{\textbf{Syntaxe}}
${\color{Green}\backslash psset}\{{\color{Green}linewidth}=2.5pt\}$
\end{block}
\end{itemize}
}
% % % % % % % % % % % % % % % % % % % % % % % % % % % %
\frame{\frametitle{}
\textbf{Les unités utilisable sont :}
\begin{itemize}
\item \textbf{millimètre :} mm
\item \textbf{centimètre :} cm
\item \textbf{pouce :} in
\item \textbf{point :} pt (0,35 mm)
\item \textbf{largeur d'un $m$ :} em
\item \textbf{hauteur d'un $x$ :} ex
\end{itemize}
}

\section{Grilles}
\frame{\frametitle{\textbf{\huge{Grilles :}}}
On peut dessiner des grilles. 
\begin{block}{\textbf{Syntaxe}}
${\color{Green}\backslash psgrid} (x_1, y_1)(x_2, y_2)$
\end{block}
\pause
\begin{block}{\textbf{Code source :}}
$\backslash begin{\color{red}\{pspicture\}}(-1,-1)(2,1)\\
{\color{Green}\backslash psgrid}(-1, -1)(2, 1)\\
\backslash end{\color{red}\{pspicture\}}$
\end{block}
\begin{center}
\begin{pspicture}(-1,-1)(2,1)
\psgrid(-1,-1)(2,1)
\end{pspicture}
\end{center}
}
% % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % %
\frame{\frametitle{}
\begin{block}{\textbf{Remarque :}}
Si la commande ${\color{Green}\backslash psgrid}$ est donnée sans argument, elle
occupe toute la figure. 
\end{block}
\pause
\begin{block}{\textbf{Code source :}}
$\backslash begin{\color{red}\{pspicture\}}(-1,-1)(2,1)\\
\backslash {\color{Green}psgrid} \\
\backslash end{\color{red}\{pspicture\}}$
\end{block}
\begin{center}
\begin{pspicture}(-1,-1)(2,1)
\psgrid(-1,-1)(2,1)
\end{pspicture}
\end{center}

}

\subsection{Options}
\frame{\frametitle{}
\textbf{Il y a plusieurs options :} 
\begin{itemize}
\item {\color{Green}unit}=${\color{orange}unit}$ \qquad par défaut : ${\color{orange}1cm}$\\
Longueur du côté du carreau unité.
\pause
\item {\color{Green}gridwidth}=${\color{orange}dim}$ \qquad par défaut : ${\color{orange}0.8pt}$\\
Épaisseur des lignes principales
\pause
\item {\color{Green}gridcolor}=${\color{orange}color}$ \qquad  par défaut : {\color{orange}black}\\
Couleur des lignes principales.
\pause
\item {\color{Green}griddots}=${\color{orange}num}$ \qquad par défaut : ${\color{orange}0}$ \\
Si $num > 0$, les lignes sont en pointillés, avec $num$
points par graduation.
\pause
\item {\color{Green}gridlabels}=${\color{orange}dim}$ \qquad  par défaut : ${\color{orange}10pt}$\\
Taille des nombres utilisés pour graduer.
\pause
\item {\color{Green}gridlabelcolor}=${\color{orange}color}$ \qquad par défaut : {\color{orange}black}\\
Couleur des nombres utilisés pour graduer.
\pause
\item {\color{Green}subgriddiv}=${\color{orange}int}$ \qquad  par défaut : ${\color{orange}5}$\\
Nombre de sous-graduations de la grille.
\end{itemize}
}
% % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % 
\frame{\frametitle{}
\begin{itemize}
\item {\color{Green}subgridwidth}=${\color{orange}dim}$ \qquad par défaut : ${\color{orange}0.4pt}$\\
Épaisseur des lignes secondaires.
\pause
\item {\color{Green}subgridcolor}=${\color{orange}color}$ \qquad  par défaut : {\color{orange}gray}\\
Couleur des lignes secondaires.
\pause
\item {\color{Green}subgriddots}=${\color{orange}num}$ \qquad par défaut : ${\color{orange}0}$\\
Comme $griddots$, pour les sous-graduations.
\end{itemize}
}

\section{Options}
\frame{\frametitle{\textbf{\huge{Options :}}}
{\color{blue}[opt]} est une option qui permet de modifier les caractéristiques des objets représentés.
\begin{itemize}
\item \textbf{\'{E}paisseur :} ${\color{blue}[\text{linewidth=largeur}]}$ modifie l'épaisseur de la ligne.
\pause
\item \textbf{Style :} pointillés, $\cdots$ \\
${\color{blue}[\text{linestyle=dashed}]}$ trace la ligne en traitillés et\\
${\color{blue}[\text{linestyle=dotted}]}$, en pointillés.
On peut combiner les deux.
\item \textbf{ Couleur :} 
${\color{blue}[\text{linecolor=couleur}]}$ pour colorier l'objet avec une couleur.\\
${\color{blue}[\text{fillstyle=solid}]}$ permet de remplir l'objet par la couleur sollicitée au moyen de la commande ${\color{blue}[\text{fillcolor=couleur}]}$.
\pause
\item \textbf{Points :} On peut changer le style d'affichage des points (par défaut, un disque noir).\\
Les $10$ styles sont :\\
{\color{blue}(square, square*, *, triangle, triangle*, diamond, diamond*, pentagon, pentagon*)}.
\pause
\item \textbf{Longueur des fl\`eches :} ${\color{blue}[\text{arrowlength=longueur}]}$ permet de changer la longueur des flèches.
\end{itemize}
}
% % % % % % % % % % % % % % % % % % % % % % % % % % % % % %

\item \textbf{{\color{blue}hatchangle}}=angle des traits.
\end{itemize}
\end{itemize}
}
